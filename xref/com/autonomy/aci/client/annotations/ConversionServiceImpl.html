11  import java.util.concurrent.ConcurrentHashMap;
12  import java.util.concurrent.ConcurrentMap;

14  /**
15   * Default implementation of ConversionService.
16   */
17  class ConversionServiceImpl implements ConversionService {
18  
19      private final ConcurrentMap<ClassPair, Converter<?, ?>> converters = new ConcurrentHashMap<>();
20  
21      public ConversionServiceImpl() {
22          registerConverter(new StringToDoubleConverter(), String.class, Double.class);
23          registerConverter(new StringToDoubleConverter(), String.class, Double.TYPE);
24  
25          registerConverter(new StringToByteConverter(), String.class, Byte.class);
26          registerConverter(new StringToByteConverter(), String.class, Byte.TYPE);
27  
28          registerConverter(new StringToIntegerConverter(), String.class, Integer.class);
29          registerConverter(new StringToIntegerConverter(), String.class, Integer.TYPE);
30  
31          registerConverter(new StringToFloatConverter(), String.class, Float.class);
32          registerConverter(new StringToFloatConverter(), String.class, Float.TYPE);
33  
34          registerConverter(new StringToShortConverter(), String.class, Short.class);
35          registerConverter(new StringToShortConverter(), String.class, Short.TYPE);
36  
37          registerConverter(new StringToCharacterConverter(), String.class, Character.class);
38          registerConverter(new StringToCharacterConverter(), String.class, Character.TYPE);
39  
40          registerConverter(new StringToBooleanConverter(), String.class, Boolean.class);
41          registerConverter(new StringToBooleanConverter(), String.class, Boolean.TYPE);
42  
43          registerConverter(new StringToLongConverter(), String.class, Long.class);
44          registerConverter(new StringToLongConverter(), String.class, Long.TYPE);
45      }
46  
47      @Override
48      public <S, T> boolean canConvert(final Class<S> inClass, final Class<T> outClass) {
49          final ClassPair classPair = new ClassPair(inClass, outClass);
50  
51          return converters.containsKey(classPair);
52      }
53  
54      @Override
55      public <S, T> T convert(final S in, final Class<S> inClass, final Class<T> outClass) {
56          final ClassPair classPair = new ClassPair(inClass, outClass);
57  
58          @SuppressWarnings("unchecked")
59          // converters are only registered in the constructor where they are correctly registered
60          final Converter<S, T> converter = (Converter<S, T>) converters.get(classPair);
61          return converter.convert(in);
62      }
63  
64      @Override
65      public <S, T> void registerConverter(final Converter<S, T> converter, final Class<S> inClass, final Class<T> outClass) {
66          final ClassPair classPair = new ClassPair(inClass, outClass);
67          converters.put(classPair, converter);
68      }
69  
70      private static class ClassPair {
71          @SuppressWarnings("unused") // Used by reflectionEquals and reflectionHashCode
72          private final Class<?> in;
73  
74          @SuppressWarnings("unused") // Used by reflectionEquals and reflectionHashCode
75          private final Class<?> out;
76  
77          private ClassPair(final Class<?> in, final Class<?> out) {
78              this.in = in;
79              this.out = out;
80          }
81  
82          @Override
83          public boolean equals(final Object o) {
84              return EqualsBuilder.reflectionEquals(this, o);
85          }
86  
87          @Override
88          public int hashCode() {
89              return HashCodeBuilder.reflectionHashCode(this);
90          }
91      }
