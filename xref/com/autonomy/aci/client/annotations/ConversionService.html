6   package com.autonomy.aci.client.annotations;
7   
8   /**
9    * Interface that defines a service for converting objects to different types.
10   */
11  interface ConversionService {
12  
13      /**
14       * Register a Converter with this ConversionService.
15       * @param converter The converter
16       * @param inClass The class of the converter's input type
17       * @param outClass The class of the converter's output type
18       * @param <S> The input type of the converter
19       * @param <T> The output type of the converter
20       */
21      <S, T> void registerConverter(Converter<S, T> converter, Class<S> inClass, Class<T> outClass);
22  
23      /**
24       * Convert the "in" object to the type S, using one of the registered Converters.
25       * @param in The input object
26       * @param inClass The class to use for the input object
27       * @param outClass The class to output
28       * @param <S> The type to use for the input object
29       * @param <T> The type to output
30       * @return The result of converting the input
31       */
32      <S, T> T convert(S in, Class<S> inClass, Class<T> outClass);
33  
34      /**
35       * Check that this ConversionService can convert from the "in" class to the "out" class.
36       * @param in The input class
37       * @param out The output class
38       * @param <S> The type represented by the input class
39       * @param <T> The type represented by the output class
40       * @return True if this ConversionService can convert from the given input class to the given output class
41       */
42      <S, T> boolean canConvert(Class<S> in, Class<T> out);
43  
44  }
