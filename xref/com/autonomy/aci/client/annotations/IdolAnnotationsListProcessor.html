7   
8   import com.autonomy.aci.client.services.AciServiceException;
9   import com.autonomy.aci.client.services.StAXProcessor;
10  import com.autonomy.aci.client.services.impl.AbstractStAXProcessor;
11  import com.autonomy.aci.client.services.impl.ErrorProcessor;

13  import javax.xml.stream.XMLStreamException;
14  import javax.xml.stream.XMLStreamReader;
15  import javax.xml.stream.events.XMLEvent;
16  import java.util.ArrayList;
17  import java.util.List;
18  import java.util.Properties;
19  
20  /**
21   * Processor that returns a list of objects of type {@code <T>}.  Intended for creation via static factory.
22   * @param <T> The type of the items in the list
23   */
24  class IdolAnnotationsListProcessor<T> extends AbstractStAXProcessor<List<T>> {
25  
26      private static final long serialVersionUID = -5340264175485205976L;
27  
28      private StAXProcessor<T> innerProcessor;
29  
30      private String rootField = AnnotationHelper.ROOT_FIELD_DEFAULT;
31      private Path rootPath = new Path("");
32  
33      <U> IdolAnnotationsListProcessor(final Class<U> clazz, final IdolAnnotationsProcessorFactory idolAnnotationsProcessorFactory, final Properties properties) {
34          final IdolDocument documentAnnotation = clazz.getAnnotation(IdolDocument.class);
35  
36          if (documentAnnotation != null) {
37              final AnnotationHelper annotationHelper = new AnnotationHelper(documentAnnotation);
38  
39              switch (documentAnnotation.fieldType()) {
40                  case PATH:
41                      rootPath = annotationHelper.getPath(properties);
42  
43                      rootField = null;
44                      break;
45                  case FIELD:
46                  default:
47                      rootField = annotationHelper.getFieldName(properties);
48              }
49          }
50  
51          setInnerProcessor(idolAnnotationsProcessorFactory.forClass(clazz));
52      }
53  
54      @Override
55      public List<T> process(final XMLStreamReader xmlStreamReader) {
56          try {
57              if (isErrorResponse(xmlStreamReader)) {
58                  this.setErrorProcessor(new ErrorProcessor());
59                  processErrorResponse(xmlStreamReader);
60              }
61  
62              final List<T> items = new ArrayList<>();
63              Path path = new Path("");
64  
65              final boolean hasRootPath = !rootPath.isEquivalent(new Path(""));
66  
67              while (xmlStreamReader.hasNext()) {
68                  final int event = xmlStreamReader.next();
69  
70                  if (event == XMLEvent.START_ELEMENT) {
71                      final String localName = xmlStreamReader.getLocalName();
72                      path = path.append(localName);
73  
74                      if (localName.equals(rootField) || (hasRootPath && path.isEquivalent(rootPath))) {
75                          final T item = innerProcessor.process(xmlStreamReader);
76                          items.add(item);
77                          path = path.removeLastComponent();
78                      }
79                  } else if (event == XMLEvent.END_ELEMENT) {
80                      if (!path.isEmpty()) {
81                          path = path.removeLastComponent();
82                      }
83                  }
84              }
85  
86              return items;
87          } catch (final XMLStreamException e) {
88              throw new AciServiceException(e);
89          }
90      }
91  
92      // private method that is only called with a ReflectionProcessor
93      @SuppressWarnings("unchecked")
94      private void setInnerProcessor(final Object processor) {
95          this.innerProcessor = (StAXProcessor<T>) processor;
96      }
97  }
