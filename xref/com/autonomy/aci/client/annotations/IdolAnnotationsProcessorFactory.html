6   package com.autonomy.aci.client.annotations;

8   import com.autonomy.aci.client.services.StAXProcessor;
9   
10  import java.util.List;
11  import java.util.Properties;
12  import java.util.ResourceBundle;
13  
14  /**
15   * Factory for producing IdolAnnotationsProcessors.
16   */
17  public interface IdolAnnotationsProcessorFactory {
18  
19      /**
20       * Returns a processor for the given class. When the {@code StAXProcessor#process} method is called the given
21       * {@code XMLStreamReader} should be on the root field for the class.
22       * @param clazz The class object for which a processor is required
23       * @param <T>   The type which for which a processor will be produced
24       * @return A {@code StAXProcessor} for {@code <T>}
25       */
26      <T> StAXProcessor<T> forClass(Class<T> clazz);
27  
28      /**
29       * Returns a processor which returns a list of objects of type {@code <T>}. This processor places no restrictions
30       * on the initial state of the {@code XMLStreamReader}.
31       * @param clazz The class object for which a processor is required
32       * @param <T>   The type which for which a processor will be produced
33       * @return A {@code StAXProcessor} for {@code List<T>}
34       */
35      <T> StAXProcessor<List<T>> listProcessorForClass(Class<T> clazz);
36  
37      /**
38       * Set the properties to use when a configured value is required via a {@code Properties} instance. Replaces existing configuration.
39       * The new properties will apply to any new processors that are created but will not apply to previously existing ones.
40       * @param properties The Properties object to use for configuration
41       */
42      void setProperties(Properties properties);
43  
44      /**
45       * Set the properties to use when a configured value is required via a {@code ResourceBundle} instance. Replaces existing configuration.
46       * The new properties will apply to any new processors that are created but will not apply to previously existing ones.
47       * <p/>
48       * RENAMED in 0.9.2 to comply with JavaBeans specification
49       * @param resourceBundle The ResourceBundle object to use for configuration
50       */
51      void setResourceBundle(ResourceBundle resourceBundle);
52  
53  }
